<UserControl x:Class="Abt.Controls.SciChart.Example.Examples.IWantTo.ZoomAndPanAChart.DragAreaToZoom"
             xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
             xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
             xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
             xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
             xmlns:sciChart="http://schemas.abtsoftware.co.uk/scichart"             
             d:DesignHeight="300"
             d:DesignWidth="500"
             Loaded="DragAreaToZoom_Loaded"
             mc:Ignorable="d">

    <Grid>
        <Grid.RowDefinitions>
            <RowDefinition Height="32" />
            <RowDefinition Height="*" />
        </Grid.RowDefinitions>

        <sciChart:SciChartSurface x:Name="sciChart"
                                  Grid.Row="1"
                                  sciChart:ThemeManager.Theme="Chrome">

            <sciChart:SciChartSurface.RenderableSeries>
                <sciChart:FastMountainRenderableSeries x:Name="mountainSeries" 
                                                       AreaBrush="#771964FF"
                                                       SeriesColor="#FF0944CF"
                                                       YAxisId="PrimaryAxis" />
                <sciChart:FastLineRenderableSeries x:Name ="lineSeries" 
                                                   SeriesColor="#FF279B27"
                                                   StrokeThickness="2"
                                                   YAxisId="SecondaryAxis" />
            </sciChart:SciChartSurface.RenderableSeries>

            <sciChart:SciChartSurface.XAxis>
                <sciChart:NumericAxis GrowBy="0.1, 0.1"/>
            </sciChart:SciChartSurface.XAxis>

            <!--  Create Y Axes on the Left and Right  -->
            <sciChart:SciChartSurface.YAxes>
                <sciChart:NumericAxis AxisAlignment="Right"
                                      Id="PrimaryAxis"
                                      TickTextBrush="#FF279B27"
                                      GrowBy="0.1, 0.1"/>
                <sciChart:NumericAxis AxisAlignment="Left"
                                      Id="SecondaryAxis"
                                      TickTextBrush="#FF4083B7" 
                                      GrowBy="0.1, 0.1"/>
            </sciChart:SciChartSurface.YAxes>

            <!--  Adding the ZoomPanModifier gives SciChart the ability to pan on mouse-drag  -->
            <sciChart:SciChartSurface.ChartModifier>
                <sciChart:ModifierGroup>
                    <sciChart:RubberBandXyZoomModifier x:Name="rubberBandZoomModifier"
                                                       IsEnabled="True"
                                                       IsXAxisOnly="False" 
                                                       ZoomExtentsY="False"
                                                       IsAnimated="True"/>
                    <sciChart:ZoomExtentsModifier x:Name="zoomExtentsModifier" 
                                                  ExecuteOn="MouseDoubleClick" />
                </sciChart:ModifierGroup>
            </sciChart:SciChartSurface.ChartModifier>

        </sciChart:SciChartSurface>

        <!--  IGNORE BELOW HERE: Toolbar to change which mouse button zoomPanModifier drags on  -->
        <StackPanel Orientation="Horizontal">
            <TextBlock Margin="3"
                       Foreground="#EEE"
                       Text="When Dragging to Zoom: "
                                           VerticalAlignment="Center"/>
            <CheckBox Margin="3"
                      Checked="ZoomXOnly_Checked"
                      Content="Zoom X-Axis Only"
                      Unchecked="ZoomXOnly_Checked"
                                          VerticalAlignment="Center"
                      ToolTipService.ToolTip="When checked, the zoom reticule expands vertically, and a zoom on the X-Axis is performed"/>

            <CheckBox Margin="3"
                      Checked="ZoomExtentsY_Checked"
                      Content="Zoom Extents Y-Axis"
                      Unchecked="ZoomExtentsY_Checked"
                                          VerticalAlignment="Center"
                      ToolTipService.ToolTip="Used in conjunction with Zoom X-Axis Only, when checked performs a Zoom to extents in the Y-direction on zoom, else leaves Y-direction unchanged"/>
            <CheckBox Margin="3"
                      Checked="UseAnimation_Checked"
                      Content="Use Animation" IsChecked="True"
                      Unchecked="UseAnimation_Checked"
                                          VerticalAlignment="Center"
                      ToolTipService.ToolTip="When checked, zoom operations are performed using animation. Disabling this feature performs an instant zoom to the new visible ranges"/>
            <Button Margin="3"
                    VerticalAlignment="Center"
                    Command="{Binding ElementName=sciChart,
                                      Path=ZoomExtentsCommand}"
                    Content="Reset Zoom" />
        </StackPanel>

    </Grid>
</UserControl>
