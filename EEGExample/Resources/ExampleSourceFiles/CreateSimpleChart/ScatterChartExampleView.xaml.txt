<UserControl x:Class="Abt.Controls.SciChart.Example.Examples.IWantTo.CreateSimpleChart.ScatterChartExampleView"
             xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
             xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
             xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
             xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
             xmlns:SciChart="http://schemas.abtsoftware.co.uk/scichart"
             xmlns:common="clr-namespace:Abt.Controls.SciChart.Example.Common"
             d:DesignHeight="300"
             d:DesignWidth="300"
             Loaded="ScatterChartExampleView_Loaded"
             mc:Ignorable="d">

    <Grid>
        <Grid.RowDefinitions>
            <RowDefinition Height="32" />
            <RowDefinition Height="*" />
        </Grid.RowDefinitions>

        <common:SciChartInteractionToolbar TargetSurface="{Binding ElementName=sciChart}" />

        <!--  Create the chart surface  -->
        <SciChart:SciChartSurface Name="sciChart"
                                  Grid.Row="1"
                                  SciChart:ThemeManager.Theme="BrightSpark">

            <!--  Declare RenderableSeries  -->
            <SciChart:SciChartSurface.RenderableSeries>
                <SciChart:XyScatterRenderableSeries x:Name="scatterRenderSeries">
                    <SciChart:XyScatterRenderableSeries.PointMarker>
                        <SciChart:EllipsePointMarker Width="9"
                                     Height="9"
                                     Fill="SteelBlue"
                                     Stroke="LightSteelBlue"
                                     StrokeThickness="2">
                        </SciChart:EllipsePointMarker>
                    </SciChart:XyScatterRenderableSeries.PointMarker>
                </SciChart:XyScatterRenderableSeries>
            </SciChart:SciChartSurface.RenderableSeries>

            <!--  Create an X Axis with Growby  -->
            <SciChart:SciChartSurface.XAxis>
                <SciChart:NumericAxis GrowBy="0.1, 0.1"/>
            </SciChart:SciChartSurface.XAxis>

            <!--  Create a Y Axis with Growby. Optional bands give a cool look and feel for minimal performance impact  -->
            <SciChart:SciChartSurface.YAxis>
                <SciChart:NumericAxis GrowBy="0.1, 0.1" DrawMajorBands="True"/>
            </SciChart:SciChartSurface.YAxis>

        </SciChart:SciChartSurface>

    </Grid>

</UserControl>
