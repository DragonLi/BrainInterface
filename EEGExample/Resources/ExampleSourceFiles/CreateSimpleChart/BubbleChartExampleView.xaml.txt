<UserControl x:Class="Abt.Controls.SciChart.Example.Examples.IWantTo.CreateSimpleChart.BubbleChartExampleView"
             xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
             xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
             xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006" 
             xmlns:d="http://schemas.microsoft.com/expression/blend/2008" 
             mc:Ignorable="d" 
             xmlns:SciChart="http://schemas.abtsoftware.co.uk/scichart"
             d:DesignHeight="300" d:DesignWidth="300">
    <Grid >

        <Grid.RowDefinitions>
            <RowDefinition Height="Auto"/>
            <RowDefinition Height="*"/>
        </Grid.RowDefinitions>
        
        <Grid.ColumnDefinitions>
            <ColumnDefinition Width="Auto"/>
            <ColumnDefinition Width="*"/>
        </Grid.ColumnDefinitions>

        <Border BorderThickness="1" BorderBrush="#EEE" Background="#FF6600" CornerRadius="3" Margin="4">
            <TextBlock Text="ZScale (Slide to change): " Foreground="#EEE" Margin="4,2,4,2"></TextBlock>
        </Border>
        
        <!-- Create the Z slider -->
        <Slider Minimum="0.5" Maximum="5.0" x:Name="zScaleSlider" Value="1.0" Margin="4,4,140,4" Orientation="Horizontal" Grid.Column="1"  HorizontalAlignment="Stretch"></Slider>
        
        <!--  Create the chart surface  -->
        <SciChart:SciChartSurface Name="sciChart" SciChart:ThemeManager.Theme="Chrome" Grid.Row="1" Grid.ColumnSpan="2">

            <!--  Declare RenderableSeries  -->
            <SciChart:SciChartSurface.RenderableSeries>
                <SciChart:FastLineRenderableSeries x:Name="lineSeries" ResamplingMode="MinMax" SeriesColor="#FFFF3333" />
                <SciChart:FastBubbleRenderableSeries x:Name="bubbleSeries" ResamplingMode="None" BubbleColor="#77333333" ZScaleFactor="{Binding ElementName=zScaleSlider, Path=Value}" AutoZRange="False" />
            </SciChart:SciChartSurface.RenderableSeries>

            <!--  Create an X Axis with Growby  -->
            <SciChart:SciChartSurface.XAxis>
                <SciChart:DateTimeAxis GrowBy="0.0, 0.1" />
            </SciChart:SciChartSurface.XAxis>

            <!--  Create a Y Axis with Growby  -->
            <SciChart:SciChartSurface.YAxis>
                <SciChart:NumericAxis GrowBy="0.1, 0.1" />
            </SciChart:SciChartSurface.YAxis>

            <SciChart:SciChartSurface.ChartModifier>
                <SciChart:ModifierGroup>
                    <SciChart:RubberBandXyZoomModifier ExecuteOn="MouseLeftButton" />
                    <SciChart:ZoomExtentsModifier ExecuteOn="MouseDoubleClick" />
                </SciChart:ModifierGroup>
            </SciChart:SciChartSurface.ChartModifier>

        </SciChart:SciChartSurface>
    </Grid>
</UserControl>
