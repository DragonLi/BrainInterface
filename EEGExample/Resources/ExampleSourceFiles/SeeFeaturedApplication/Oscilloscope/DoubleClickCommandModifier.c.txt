using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Windows;
using System.Windows.Input;
using Abt.Controls.SciChart.ChartModifiers;

namespace Abt.Controls.SciChart.Example.Examples.IWantTo.SeeFeaturedApplication.Oscilloscope
{
    public class DoubleClickCommandModifier : ChartModifierBase
    {
        public static readonly DependencyProperty CommandProperty =
            DependencyProperty.Register("Command", typeof(ICommand), typeof(DoubleClickCommandModifier), new PropertyMetadata(default(ICommand)));

        public ICommand Command
        {
            get { return (ICommand)GetValue(CommandProperty); }
            set { SetValue(CommandProperty, value); }
        }

        public override void OnModifierDoubleClick(ModifierMouseArgs e)
        {
            base.OnModifierDoubleClick(e);

            if (Command != null && Command.CanExecute(null))
                Command.Execute(null);
        }
    }
}
