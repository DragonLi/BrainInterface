using System.Windows;
using System.Windows.Controls;

namespace Abt.Controls.SciChart.Example.Examples.IWantTo.CreateStockCharts
{
    public partial class CreateRealTimeTickingStockChart : UserControl
    {
        public CreateRealTimeTickingStockChart()
        {
            InitializeComponent();
        }

        private void Aliasing_Checked(object sender, RoutedEventArgs e)
        {
            var checkbox = sender as CheckBox;
            if (checkbox != null && this.priceChart != null)
            {
                foreach(var renderableSeries in priceChart.RenderableSeries)
                {
                    renderableSeries.AntiAliasing = checkbox.IsChecked == true;
                }
                priceChart.InvalidateElement();
            }
        }
    }
}
