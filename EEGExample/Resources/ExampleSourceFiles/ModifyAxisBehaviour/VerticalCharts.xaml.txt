<UserControl x:Class="Abt.Controls.SciChart.Example.Examples.IWantTo.ModifyAxisBehaviour.VerticalCharts"
             xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
             xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
             xmlns:SciChart="http://schemas.abtsoftware.co.uk/scichart"
             xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
             xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
             Loaded="VerticalXAxis_OnLoaded"
             d:DesignHeight="300"
             d:DesignWidth="300"
             mc:Ignorable="d">

    <Grid>
        <Grid.ColumnDefinitions>
            <ColumnDefinition />
            <ColumnDefinition />
        </Grid.ColumnDefinitions>
        
        <!-- Declare a chart with two YAxis at the top. --> 
        <!-- The HorizontalChartGroup is used to synchronize axis area sizes and works like the SciChartGroup.VerticalChartGroup property, but for horizontal charts -->
        <SciChart:SciChartSurface Name="sciChartLeft"
                                  Grid.Column="0"
                                  SciChart:HorizontalGroupHelper.HorizontalChartGroup="MyHorizontalGroup"
                                  SciChart:ThemeManager.Theme="Chrome">

            <SciChart:SciChartSurface.RenderableSeries>
                <SciChart:FastLineRenderableSeries SeriesColor="Red" YAxisId="Top1" />
                <SciChart:FastLineRenderableSeries SeriesColor="Black" YAxisId="Top2" />
            </SciChart:SciChartSurface.RenderableSeries>

            <!-- Define the XAxis for this chart on the Left. Optional bands give a cool look and feel for minimal performance impact  -->
            <SciChart:SciChartSurface.XAxis>
                <SciChart:NumericAxis AxisAlignment="Left" AxisTitle="X-Axis" DrawMajorBands="True"/>
            </SciChart:SciChartSurface.XAxis>

            <SciChart:SciChartSurface.YAxes>
                <SciChart:NumericAxis AxisAlignment="Top"
                                      AxisTitle="Red Line"
                                      BorderBrush="Red"
                                      BorderThickness="0,0,0,1"
                                      Id="Top1"
                                      TickTextBrush="Red" />
                <SciChart:NumericAxis AxisAlignment="Top"
                                      AxisTitle="Black Line"
                                      BorderBrush="Black"
                                      BorderThickness="0,0,0,1"
                                      Id="Top2"
                                      TickTextBrush="Black" />
            </SciChart:SciChartSurface.YAxes>

            <SciChart:SciChartSurface.ChartModifier>
                <SciChart:ModifierGroup SciChart:MouseManager.MouseEventGroup="myCustomGroup">
                    <SciChart:RubberBandXyZoomModifier IsXAxisOnly="True" ReceiveHandledEvents="True" />
                    <SciChart:ZoomExtentsModifier />
                </SciChart:ModifierGroup>
            </SciChart:SciChartSurface.ChartModifier>
        </SciChart:SciChartSurface>

        <!-- Declare a chart with one YAxis at the top. -->
        <!-- The HorizontalChartGroup is used to synchronize axis area sizes and works like the SciChartGroup.VerticalChartGroup property, but for horizontal charts -->
        <SciChart:SciChartSurface Name="sciChartRight"
                                  Grid.Column="1"
                                  SciChart:HorizontalGroupHelper.HorizontalChartGroup="MyHorizontalGroup"
                                  SciChart:ThemeManager.Theme="Chrome">

            <SciChart:SciChartSurface.RenderableSeries>
                <SciChart:FastLineRenderableSeries SeriesColor="Green" YAxisId="Top1" />
                <SciChart:FastLineRenderableSeries SeriesColor="Blue" YAxisId="Top1" />
            </SciChart:SciChartSurface.RenderableSeries>

            <!-- Define the XAxis for this chart on the Right. Optional bands give a cool look and feel for minimal performance impact  -->
            <SciChart:SciChartSurface.XAxis>
                <SciChart:NumericAxis AxisAlignment="Right" AxisTitle="X-Axis" DrawMajorBands="True"/>
            </SciChart:SciChartSurface.XAxis>

            <SciChart:SciChartSurface.YAxes>
                <SciChart:NumericAxis AxisAlignment="Top"
                                      AxisTitle="Blue Line"
                                      BorderBrush="Blue"
                                      BorderThickness="0,0,0,1"
                                      Id="Top1"
                                      TickTextBrush="Blue" />
            </SciChart:SciChartSurface.YAxes>

            <SciChart:SciChartSurface.ChartModifier>
                <SciChart:ModifierGroup SciChart:MouseManager.MouseEventGroup="myCustomGroup">
                    <SciChart:RubberBandXyZoomModifier IsXAxisOnly="True" ReceiveHandledEvents="True" />
                    <SciChart:ZoomExtentsModifier />
                </SciChart:ModifierGroup>
            </SciChart:SciChartSurface.ChartModifier>
        </SciChart:SciChartSurface>

    </Grid>
</UserControl>
