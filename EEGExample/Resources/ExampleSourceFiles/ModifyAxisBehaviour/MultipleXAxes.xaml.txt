<UserControl x:Class="Abt.Controls.SciChart.Example.Examples.IWantTo.ModifyAxisBehaviour.MultipleXAxes"
             xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
             xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
             xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
             xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
             xmlns:s="http://schemas.abtsoftware.co.uk/scichart"
             d:DesignHeight="300"
             d:DesignWidth="300"
             Loaded="MultipleXAxes_Loaded"
             mc:Ignorable="d">

    <Grid s:ThemeManager.Theme="Chrome">
        <!--  Create the chart surface  -->
        <s:SciChartSurface Name="sciChart">

            <!--  Declare RenderableSeries and assign to Y-Axis  -->
            <s:SciChartSurface.RenderableSeries>
                <s:FastLineRenderableSeries x:Name="redLine"
                                            SeriesColor="#FFFF1919"
                                            StrokeThickness="1"
                                            XAxisId="BottomAxisId"
                                            YAxisId="LeftAxisId" />
                <s:FastLineRenderableSeries x:Name="greenLine"
                                            SeriesColor="#FF279B27"
                                            StrokeThickness="1"
                                            XAxisId="BottomAxisId"
                                            YAxisId="LeftAxisId" />
                <s:FastLineRenderableSeries x:Name="yellowLine"
                                            SeriesColor="#FFFC9C29"
                                            StrokeThickness="1"
                                            XAxisId="TopAxisId"
                                            YAxisId="RightAxisId" />
                <s:FastLineRenderableSeries x:Name="blueLine"
                                            SeriesColor="#FF4083B7"
                                            StrokeThickness="1"
                                            XAxisId="TopAxisId"
                                            YAxisId="RightAxisId" />
            </s:SciChartSurface.RenderableSeries>

            <!--  Create an X Axis  -->
            <s:SciChartSurface.XAxes>
                <s:DateTimeAxis AxisAlignment="Bottom"
                                AxisTitle="Bottom Axis"
                                BorderBrush="#FFFF1919"
                                BorderThickness="0,1,0,0"
                                Id="BottomAxisId"
                                TickTextBrush="#FFFF1919"/>
                <s:DateTimeAxis AxisAlignment="Top"
                                AxisTitle="Top Axis"
                                BorderBrush="#FF222222"
                                BorderThickness="0,0,0,1"
                                Id="TopAxisId"
                                TickTextBrush="#FF222222" />
            </s:SciChartSurface.XAxes>

            <!--  Create Y Axes on the Left and Right. Optional bands give a cool look and feel for minimal performance impact  -->
            <s:SciChartSurface.YAxes>
                <s:NumericAxis AxisAlignment="Left"
                               AxisTitle="Left Axis"
                               BorderBrush="#FFFF1919"
                               BorderThickness="0,0,1,0"
                               GrowBy="0.05, 0.05"
                               Id="LeftAxisId"
                               TextFormatting="#.0"
                               TickTextBrush="#FFFF1919" 
                               DrawMajorBands="True"/>
                <s:NumericAxis AxisAlignment="Right"
                               AxisTitle="Right Axis"
                               BorderBrush="#FF222222"
                               BorderThickness="1,0,0,0"
                               GrowBy="0.05, 0.05"
                               Id="RightAxisId"
                               TextFormatting="#.0"
                               TickTextBrush="#FF222222"/>
            </s:SciChartSurface.YAxes>

            <!--  Create chart modifiers to zoom, drag X,Y Axes and double click to zoom extents  -->
            <s:SciChartSurface.ChartModifier>
                <s:ModifierGroup>
                    <s:RubberBandXyZoomModifier IsXAxisOnly="True" />
                    <s:XAxisDragModifier AxisId="TopAxisId" />
                    <s:XAxisDragModifier AxisId="BottomAxisId" />
                    <s:YAxisDragModifier AxisId="LeftAxisId" />
                    <s:YAxisDragModifier AxisId="RightAxisId" />
                    <s:ZoomPanModifier ExecuteOn="MouseRightButton" />
                    <s:ZoomExtentsModifier ExecuteOn="MouseDoubleClick" />
                    <s:LegendModifier ShowLegend="True" GetLegendDataFor="AllSeries" Margin="10"/>
                </s:ModifierGroup>
            </s:SciChartSurface.ChartModifier>

        </s:SciChartSurface>

    </Grid>
</UserControl>
