<UserControl x:Class="Abt.Controls.SciChart.Example.Examples.IWantTo.StyleAChart.ChangeTheme"
             xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
             xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
             xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
             xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
             xmlns:sciChart="http://schemas.abtsoftware.co.uk/scichart"
             d:DesignHeight="300"
             d:DesignWidth="300"
             Loaded="ChangeTheme_Loaded"
             mc:Ignorable="d">
    <Grid>
        <Grid.RowDefinitions>
            <RowDefinition Height="32" />
            <RowDefinition Height="*" />
            <RowDefinition Height="32" />
        </Grid.RowDefinitions>

        <sciChart:SciChartSurface x:Name="sciChart"
                                  Grid.Row="1"
                                  ChartTitle="Chart Title"
                                  sciChart:ThemeManager.Theme="BlackSteel">

            <!--  Add some sample RenderableSeries. Each theme has a default style for these  -->
            <sciChart:SciChartSurface.RenderableSeries>
                <sciChart:FastMountainRenderableSeries x:Name="mountainSeries" YAxisId="PrimaryAxisId" />
                <sciChart:FastCandlestickRenderableSeries x:Name="candlestickSeries"
                                                          AntiAliasing="False"
                                                          YAxisId="PrimaryAxisId" />
                <sciChart:FastLineRenderableSeries x:Name="lineSeries" YAxisId="PrimaryAxisId" />
                <sciChart:FastColumnRenderableSeries x:Name="columnSeries" YAxisId="SecondaryAxisId" />
            </sciChart:SciChartSurface.RenderableSeries>

            <!--  Add a DateTime XAxis  -->
            <sciChart:SciChartSurface.XAxis>
                <sciChart:DateTimeAxis x:Name="xAxis" AxisTitle="Time Axis Title" />
            </sciChart:SciChartSurface.XAxis>

            <!--  Add two YAxes, to display price and volume data  -->
            <sciChart:SciChartSurface.YAxes>

                <sciChart:NumericAxis AxisAlignment="Right"
                                      AxisTitle="Axis Right Title"
                                      DrawMajorBands="True"
                                      Id="PrimaryAxisId"
                                      TextFormatting="#.0">
                    <sciChart:NumericAxis.GrowBy>
                        <!--
                            Using GrowBy of 10% below and 20% above, has the effect of
                            aligning all series on this axis just below centre
                        -->
                        <sciChart:DoubleRange Max="0.1" Min="0.20" />
                    </sciChart:NumericAxis.GrowBy>
                </sciChart:NumericAxis>

                <sciChart:NumericAxis AxisAlignment="Left"
                                      AxisTitle="Axis Left Title"
                                      Id="SecondaryAxisId"
                                      TextFormatting="###E+0">
                    <sciChart:NumericAxis.GrowBy>
                        <!--
                            Using GrowBy of 0% below and 300% above, has the effect of
                            aligning all series on this axis to the bottom quarter of the axis
                        -->
                        <sciChart:DoubleRange Max="3.0" Min="0" />
                    </sciChart:NumericAxis.GrowBy>
                </sciChart:NumericAxis>


            </sciChart:SciChartSurface.YAxes>

            <!--  Add some modifiers to zoom, zoom extents  -->
            <sciChart:SciChartSurface.ChartModifier>
                <sciChart:ModifierGroup>
                    <sciChart:XAxisDragModifier></sciChart:XAxisDragModifier>
                    <sciChart:YAxisDragModifier AxisId="SecondaryAxisId"></sciChart:YAxisDragModifier>
                    <sciChart:YAxisDragModifier AxisId="PrimaryAxisId"></sciChart:YAxisDragModifier>
                    <sciChart:LegendModifier ShowLegend="True" ShowVisibilityCheckboxes="False" Margin="10" />
                    <sciChart:RubberBandXyZoomModifier />
                    <sciChart:ZoomPanModifier ExecuteOn="MouseRightButton" />
                    <sciChart:ZoomExtentsModifier ExecuteOn="MouseDoubleClick" />
                    <sciChart:CursorModifier ReceiveHandledEvents="True"
                                             ShowAxisLabels="True"
                                             ShowTooltip="True"
                                             ShowTooltipOn="MouseRightButtonDown" />
                </sciChart:ModifierGroup>
            </sciChart:SciChartSurface.ChartModifier>

        </sciChart:SciChartSurface>

        <sciChart:SciChartOverview x:Name="overview"
                                   Grid.Row="2"
                                   sciChart:ThemeManager.Theme="ExpressionLight"
                                   SelectedRange="{Binding ElementName=sciChart,
                                                           Path=XAxis.VisibleRange,
                                                           Mode=TwoWay}" />

        <StackPanel Orientation="Horizontal">
            <TextBlock Margin="5,5"
                       Foreground="#EEE"
                       Text="Update Theme: " />
            <ComboBox x:Name="cboTheme"
                      MinWidth="140"
                      Margin="5,3"
                      SelectionChanged="cboThemeChanged" />
        </StackPanel>
    </Grid>
</UserControl>
