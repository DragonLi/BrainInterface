namespace Abt.Controls.SciChart.Example.Examples.IWantTo.CreateSimpleChart
{
    /// <summary>
    /// Interaction logic for StackedColumnChartSideBySideExampleView.xaml
    /// </summary>
    public partial class StackedColumnChartSideBySideExampleView : UserControl
    {
        public StackedColumnChartSideBySideExampleView()
        {
            InitializeComponent();

            var rnd = new Random();
            var dataSeries1 = new XyDataSeries<double, double>() { SeriesName = "data1" };
            var dataSeries2 = new XyDataSeries<double, double>() { SeriesName = "data2" };
            var dataSeries3 = new XyDataSeries<double, double>() { SeriesName = "data3" };
            var dataSeries4 = new XyDataSeries<double, double>() { SeriesName = "data4" };
            const int n = 30;
            for (int i = 0; i < n; i++)
            {
                dataSeries1.Append(i, rnd.NextDouble());
                dataSeries2.Append(i, rnd.NextDouble());
                dataSeries3.Append(i + 0.4, rnd.NextDouble());
                dataSeries4.Append(i + 0.4, rnd.NextDouble());
            }

            using (this.sciChart.SuspendUpdates())
            {
                this.columnSeries1.DataSeries = dataSeries1;
                this.columnSeries2.DataSeries = dataSeries2;
                this.columnSeries3.DataSeries = dataSeries3;
                this.columnSeries4.DataSeries = dataSeries4;
            }
        }
    }
}
